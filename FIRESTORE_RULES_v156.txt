// v156 - stałe przesyłki zapisują request osobno; profil klienta = e-mail
rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    function signedIn() {
      return request.auth != null && request.auth.token.email != null;
    }
    function email() {
      return request.auth.token.email;
    }
    function isAdmin() {
      return signedIn() && email() == "contact@mkulczyk.pl";
    }

    match /admin/{docId} {
      allow read, write: if isAdmin();
    }

    match /settings/{docId} {
      allow read, write: if isAdmin();
    }

    match /clients/{clientEmail} {
      allow read, write, delete: if isAdmin();
      allow read: if signedIn() && clientEmail == email();
      allow update: if signedIn()
        && clientEmail == email()
        && resource.data.email == email()
        && request.resource.data.email == email()
        && request.resource.data.role == "client";
      allow create, delete: if false;
    }

    match /requests/{requestId} {
      allow read, write, delete: if isAdmin();
      allow read: if signedIn() && resource.data.clientEmail == email();
      allow create: if signedIn()
        && request.resource.data.clientEmail == email()
        && request.resource.data.ownerEmail == email();
      allow update: if signedIn()
        && resource.data.clientEmail == email()
        && request.resource.data.clientEmail == email()
        && request.resource.data.ownerEmail == email();
    }

    match /notifications/{notificationId} {
      allow read, write, delete: if isAdmin();
      allow create, update: if signedIn()
        && request.resource.data.targetEmail == "contact@mkulczyk.pl"
        && request.resource.data.clientEmail == email();
    }
  }
}
